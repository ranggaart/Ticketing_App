<?php

namespace App\Http\Controllers\Admin;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use App\Models\Lokasi;

class LokasiController extends Controller
{
    /**
     * Menampilkan daftar semua lokasi.
     */
    public function index()
    {
        // Mengambil seluruh data lokasi
        $lokasis = Lokasi::all();

        // Menampilkan halaman daftar lokasi
        return view(
            'admin.lokasi.index',
            compact('lokasis')
        );
    }

    /**
     * Menampilkan form tambah lokasi.
     */
    public function create()
    {
        return view('admin.lokasi.create');
    }

    /**
     * Menyimpan lokasi baru.
     */
    public function store(Request $request)
    {
        // Validasi input
        $validatedData = $request->validate([
            'nama'   => 'required|string|max:255',
            'alamat' => 'required|string',
        ]);

        // Simpan ke database
        Lokasi::create($validatedData);

        // Kembali ke halaman index
        return redirect()
            ->route('admin.lokasis.index')
            ->with('success', 'Lokasi berhasil ditambahkan.');
    }

    /**
     * Menampilkan detail lokasi.
     */
    public function show(string $id)
    {
        // Ambil data lokasi
        $lokasi = Lokasi::findOrFail($id);

        return view(
            'admin.lokasi.show',
            compact('lokasi')
        );
    }

    /**
     * Menampilkan form edit lokasi.
     */
    public function edit(string $id)
    {
        // Ambil data lokasi
        $lokasi = Lokasi::findOrFail($id);

        return view(
            'admin.lokasi.edit',
            compact('lokasi')
        );
    }

    /**
     * Memperbarui data lokasi.
     */
    public function update(Request $request, string $id)
    {
        try {

            // Ambil lokasi
            $lokasi = Lokasi::findOrFail($id);

            // Validasi
            $validatedData = $request->validate([
                'nama'   => 'required|string|max:255',
                'alamat' => 'required|string',
            ]);

            // Update data
            $lokasi->update($validatedData);

            return redirect()
                ->route('admin.lokasis.index')
                ->with('success', 'Lokasi berhasil diperbarui.');

        } catch (\Exception $e) {

            return redirect()
                ->back()
                ->withErrors([
                    'error' => 'Terjadi kesalahan saat memperbarui lokasi: ' . $e->getMessage()
                ]);
        }
    }

    /**
     * Menghapus lokasi.
     */
    public function destroy(string $id)
    {
        // Ambil lokasi
        $lokasi = Lokasi::findOrFail($id);

        // Cek apakah masih digunakan oleh event
        if ($lokasi->events()->count() > 0) {

            return redirect()
                ->route('admin.lokasis.index')
                ->with(
                    'error',
                    'Lokasi tidak dapat dihapus karena masih digunakan oleh event.'
                );
        }

        // Hapus lokasi
        $lokasi->delete();
        return redirect()
            ->route('admin.lokasis.index')
            ->with('success', 'Lokasi berhasil dihapus.');
    }
}
